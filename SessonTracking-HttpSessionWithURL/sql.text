CREATE TABLE "SYSTEM"."JOB_TABLE" 
   (	"JOBID" NUMBER(*,0) NOT NULL ENABLE, 
	"NAME" VARCHAR2(20 BYTE) NOT NULL ENABLE, 
	"AGE" VARCHAR2(20 BYTE) NOT NULL ENABLE, 
	"ADDRESS" VARCHAR2(20 BYTE) NOT NULL ENABLE, 
	"YEAR_OF_EXPERIENCE" VARCHAR2(20 BYTE), 
	"SKILL" VARCHAR2(20 BYTE) NOT NULL ENABLE, 
	"DESIRED_CITY" VARCHAR2(20 BYTE), 
	"EXPECTED_SALARY" VARCHAR2(20 BYTE), 
	 CONSTRAINT "JOB_TABLE_PK" PRIMARY KEY ("JOBID")
